\begin{enumerate}

\item Strom $\underline{I}$ entspricht der Summe von $\underline{I}_{RC}$ und $\underline{I}_L$.

\item Achsen neu bestimmen: Re in Richtung $\underline{U}_L$. 

\end{enumerate}
\end{multicols}

In Produktform \hspace{0.5cm}
\parbox{3cm}{\[ F(\omega ) = K_1 \prod_{i=1} ^r \mbox{Standardterme,} \;\;\; K_1 = \frac{a_0}{b_0} \] }\\
\mbox{\hspace{1cm} }Standardterme \hspace{0.5cm}  \parbox{8cm}{\[ \left\{ \begin{array}{l}
	                                 (j \omega T ) ^n \\
	                                 (1 + j \omega T ) ^n \\
	                                 \left [ 1 + 2 \xi j \omega T + (j \omega ) ^2 T^2 \right] ^n
                                         \end{array} \right. \]}
$ n \pm 1$ , $ \pm 2 $ ... \\
\mbox{\hspace{4cm} }
\parbox{5cm}{\[ 1 + 2 \xi j \omega T + (j \omega ) ^2 T^2 = \left\{ \begin{array}{ll}
	(1 + j \omega T_1) (1 + j \omega T_2) & \mbox{für} \quad \xi > 1 \\ 
	(1 + j \omega T ) ^2 & \mbox{für} \quad \xi = 1 \\
	\mbox{nicht aufspaltbar} & \mbox{für} \quad \xi < 1 \\
	\end{array} \right. \]} \\
Normierung
\begin{tabbing}
$\bullet $ Frequenz: \= Bezugsfrequenz $ \omega_0 = \frac{1}{T_0} $ $ \Longrightarrow $ normierte Frequenz $ \Omega = \frac{\omega}{\omega_0} = \omega T_0 $ \\

\end{tabbing}
