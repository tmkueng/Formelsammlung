\section{Drehstrom Synchrongenerator (DSG)} 

\index{Generator} 
\index{Synchrongenerator}


\Hauptbox{
 \Bildbox{
	\input{eat/DSG/DSG.pstex_t}
	\begin{scriptsize}I, IV: Motorbetrieb\\
	II,III: Generatorbetrieb\\
	I,II: Abgabe induktive Blindleistung $=$ Übererregt\\
	III, IV: Aufnahme kapazitive Blindleistung $=$ Untererregt\end{scriptsize}
}
 \Formelbox{
	\[ n_{syn} = \frac{60 f}{p} \]
	\[ U_i = z \frac{d \Phi}{dt} \]
	\[ |U_i| = B l v_R z \]
	\[ I_w = I \cos(\varphi) \mbox{\hspace{0.5cm}} I_b = I \sin(\varphi) \]
	\[ I^2 = I_w^2 + I_b^2 \]
	\[ \underline{I} = I_w + j I_b  = \frac{j}{X_d} (\underline{U}_p - \underline{U}_{Kl})\] 
	\[ \underline{U}_{Kl} = \underline{U}_p + j X_d \underline{I} \]
	\[ X_d = X_H + X_{\sigma} \]
	\begin{footnotesize}Leerlauf:\end{footnotesize}	\[ \frac{I_E}{I_{E0N}} = \frac{U_p \sqrt{3}}{U_N} \]
	\begin{footnotesize}Kurzsschluss:\end{footnotesize} \[X_d = \frac{U_p}{I_{K0}} \]
	\[ x_d = X_d \frac{I_N \sqrt{3}}{U_N} = X_d \frac{I_N}{U_{Kl}} = \frac{1}{k_0} \]
  }}{
 \Groessenbox{
  $n_{syn}$ & Drehzahl                 & $[\frac{1}{min}]$ \\
  $f$       & (Netz-) Frequenz         & $[Hz]$ \\
  $p$       & Polpaarzahl              & $[1]$ \\
  $U_i$     & Span. induziert          & $[V]$ \\
  $U_{Kl}$  & Span. Klemmen            & $[V]$ \\
  $l$       & Leiterlänge              & $[m]$\\
  $z$       & Anz. Windungen           & $[1]$ \\
  $v_R$     & Luftspaltgeschw.         & $[\frac{m}{s}]$\\
  $B$       & Induktion                & $[\frac{Vs}{m^2}]$\\
  $\Phi$    & magn. Fluss              & $[Wb]$\\
  $I_w $    & Wirkstrom                & $[A]$\\
  $I_b $    & Blindstrom               & $[A]$\\
  $X_d$     & synch.Reakt              & $[\Omega]$\\
  $x_d$     & relative synch.Reakt     & $[\Omega]$\\
  $I_E$     & Erregerstrom             & $[A]$\\
  $ U_N$    & U-Nenn verkettet         & $[V] $\\
  $ k_0$    & Leerlauf- Kurzschluss Verhältnis         & $[1] $\\

	\[ U_p \sin(\vartheta) = I X_d \cos (\varphi ) \]
