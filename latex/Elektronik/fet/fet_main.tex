\index{Fet!MOSFET!ideal} 
\Hauptbox{                                                                    
  \Bildbox{
\begin{footnotesize}Im gesättigten Bereich verhält sich ein FET annähernd wie eine Stromquelle, im ungesättigten Bereich stellt er einen Widerstand dar. Die Steuergrösse ist $ u_{GS}$.\end{footnotesize} \\ \\
	\input{Elektronik/fet/kennlinieNFet.pstex_t}
	\begin{itemize}
	\item Drainstrom fliesst nur falls $|u_{GS}| > |U_{T}|$.
	\item Gatestrom ist 0.
	\end{itemize} 
  }
  \Formelbox{
	\[ U_{DSsat} = U_{GS} - U_{T} \]
	\[ I_{Dsat} = K \frac{U_{DS} ^2}{2} \]
	\[ I_D = I_D' \frac{W}{L} \]
	\[ K = \frac{2 I_{DSS} ^*}{U_{T} ^2} \]
	\[ K = K' \frac{W}{L} \qquad \qquad K' = \mu C_{ox} \]
	\[ k' \approx K' \qquad \qquad C_{ox} = \frac{\epsilon_{ox}}{t_{ox}} \]
	\[ U_A \approx a L \]
        \begin{footnotesize}Im ohmschen Bereich gilt:\end{footnotesize}
	\[ U_{DS} < U_{DSsat} \]
	\[ I_D = K \left[ (U_{GS} - U_{T} ) U_{DS} - \frac{U ^2 _{DS}}{2} \right ] \]
	\[r_{DS} = \frac{dV_{DS}}{dI_D} = \frac{|U_A| + U_{DS}}{I_D} \]
        \begin{footnotesize}Im gesättigten Bereich gilt:\end{footnotesize}
	\[ U_{DS} > U_{DSsat} \]
	\[ I_D = \frac{k}{2} (U_{GS} - U_{T}) ^2 \]
        \begin{footnotesize}Nur bei n-Kanal:\end{footnotesize}
	\[ U_{T} > 0 \]
	\[ U_{GS} > U_{T} \Rightarrow I_D > 0 \]
        \begin{footnotesize}Nur bei p-Kanal:\end{footnotesize}
	\[ U_{T} < 0 \]
	\[ U_{GS} < U_{T} \Rightarrow I_D > 0 \]
      } 
  }

	\begin{footnotesize}gesättigt: $ U_{DS} \geq U_{DSsat} $\end{footnotesize}

	\[ \Delta U_T = \gamma ( \sqrt{U_{SB} \pm \Phi_0} \sqrt{\Phi_0}) \]
	\[ U_{temp} = \frac{kT}{e} = 86\frac{\mu U}{K}T \]
	\[ I_M = I_M' \frac{W}{L} \]
	\[ n = 1 + \frac{\gamma}{2 \sqrt{U_{SB} + \Phi_0}}  \]
	\[ K = K' \frac{W}{L} = \mu C_{oc} \frac{W}{L} \]
	\[ k = k'\frac{W}{L} = \mu C_{ox} \frac{W}{L}, \qquad \alpha = \frac{K'}{k'} \]

	\[ g_m = K (U_{GS} - U_{T} ) ( 1 + \lambda U_{DS} ) \]
	\[ g_m = \sqrt{2 k I_D (1 + \lambda U_{DS} ) }  \]
	\[ g_m \thickapprox \sqrt{2 k I_D}\]
	\[ \frac{g_{m1}}{g_{m2}} = \sqrt{\frac{I_{D1}} {I_{D2}}} \]
	\[ r_S = \frac{1}{g_m} \]

	\[ r_{DS} = \frac{U_A + |U_{DS}|}{| I_D | } \thickapprox \frac{| U_A |}{| I_D | }  \]

        \[ g_m = \frac{dI_D}{dU_{GS}} = \frac{I_D}{n U_{temp}} \]
	\[ r_S = \frac{1}{g_m} = \frac{n U_{temp}}{I_D} \]

	\[ g_{mB} = -g_m \frac{\gamma}{2 \sqrt{U_{SB} + \Phi _0}} \]
	\[ g_{mB} = -g_{mB}(n-1) \]
	\begin{small}$\Rightarrow $ Back-Gate hat die halbe Wirkung des Gate bei $U_{SB} = 0$\end{small}
      }
  }

	\begin{footnotesize}\[ \left | \begin{array}{lcl}
			u_G - i_D R_S -u_{GS} & = & 0 \\
			U^+ - i_D (R_S + R_D ) - u_{DS} & = & 0 \\
		\end{array} \right | \]\end{footnotesize}

	\begin{small}\[ \left | \begin{array}{lcl}
			u_G - \frac{R_S U^+}{R_S + R_D + r_{DS0}} - u_{GS} & = & 0 \\
			r_{DS0} = \frac{1}{K (U_{GS} - U_{T} )} & = & 0 \\
		\end{array} \right | \]\end{small}
      }
  }
